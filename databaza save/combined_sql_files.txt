create table dochadzky
(
    id_dochadza   int auto_increment
        primary key,
    ziak_id       int  not null,
    id_rozvrh     int  not null,
    datum         date not null,
    typ           int  not null,
    poznamka      int  null,
    ospravedlnene int  null,
    constraint dochadzky_osoba_osoba_id_fk
        foreign key (ziak_id) references osoba (osoba_id),
    constraint dochadzky_rozvrhy_rozvrh_id_fk
        foreign key (id_rozvrh) references rozvrhy (rozvrh_id)
);

create definer = root@`%` trigger after_dochadzka_insert
    after insert
    on dochadzky
    for each row
begin
    declare skratkaPredmetu VARCHAR(10);
    declare info VARCHAR(255);
    declare dlzkaMeskania VARCHAR(255);
    select z.skratka into skratkaPredmetu
    from rozvrhy r
    join predmety p on r.predmet_id = p.predmet_id
    join zoznam_predmetov z on p.zoznam_predmet_id = z.zoznam_predmet_id
    where r.rozvrh_id = NEW.id_rozvrh;
    if NEW.typ = 1 then
        set info = CONCAT('Chýbanie na predmete ', skratkaPredmetu);
        if new.ospravedlnene = 1 then
           set info = CONCAT('Chýbanie na predmete ', skratkaPredmetu, '- Ospravedlnené');
        elseif NEW.ospravedlnene = 2 then
            set info = CONCAT('Chýbanie na predmete ', skratkaPredmetu, '- Neospravedlnené');
        end if;
    elseif NEW.typ = 2 then
        set dlzkaMeskania = IFNULL(NEW.poznamka, 'neuvedená');
        set info = CONCAT('Meškanie na predmete ', skratkaPredmetu, ', dĺžka meškania: ', dlzkaMeskania);
        if new.ospravedlnene = 1 then
           set info = CONCAT('Meškanie na predmete ', skratkaPredmetu, ', dĺžka meškania: ', dlzkaMeskania, '- Ospravedlnené');
        elseif NEW.ospravedlnene = 2 then
            set info = CONCAT('Meškanie na predmete ', skratkaPredmetu, ', dĺžka meškania: ', dlzkaMeskania, '- Neospravedlnené');
        end if;
    end if;
    insert into oznamenia(osoba_id, datum, sprava)
    values (NEW.ziak_id, NOW(), info);
END;

create definer = root@`%` trigger after_dochadzka_update
    after update
    on dochadzky
    for each row
BEGIN
    DECLARE skratkaPredmetu VARCHAR(10);
    DECLARE info VARCHAR(255);
    DECLARE ospravedlnenieSprava VARCHAR(255);
    select z.skratka into skratkaPredmetu
    from rozvrhy r
    join predmety p on r.predmet_id = p.predmet_id
    join zoznam_predmetov z on p.zoznam_predmet_id = z.zoznam_predmet_id
    where r.rozvrh_id = NEW.id_rozvrh;

    IF NEW.typ != OLD.typ THEN
        IF NEW.typ = 2 THEN
            SET info = CONCAT('Zmena z Chýbania na Meškanie na predmete ', skratkaPredmetu, ', čas meškania: ', new.poznamka);
        ELSEIF NEW.typ = 1 THEN
            SET info = CONCAT('Zmena z Meškania na Chýbanie na predmete ', skratkaPredmetu);
        END IF;
    else
        IF NEW.typ = 2 THEN
            SET info = CONCAT('Meškanie na predmete ', skratkaPredmetu, ', čas meškania: ', new.poznamka);
        ELSEIF NEW.typ = 1 THEN
            SET info = CONCAT('Chýbanie na predmete ', skratkaPredmetu);
        END IF;
    end if;
    IF NEW.ospravedlnene = 1 THEN
        SET ospravedlnenieSprava = ' - Ospravedlnené';
    ELSEIF NEW.ospravedlnene = 2 THEN
        SET ospravedlnenieSprava = ' - Neospravedlnené';
    ELSE
        SET ospravedlnenieSprava = '';
    END IF;
    INSERT INTO oznamenia(osoba_id, datum, sprava)
    VALUES (NEW.ziak_id, NOW(), concat(info, ospravedlnenieSprava));
END;

create definer = root@`%` trigger before_dochadzka_delete
    before delete
    on dochadzky
    for each row
BEGIN
    DECLARE skratkaPredmetu VARCHAR(10);
    DECLARE sprava VARCHAR(255);
    SELECT z.skratka INTO skratkaPredmetu
    FROM rozvrhy r
    JOIN predmety p ON r.predmet_id = p.predmet_id
    JOIN zoznam_predmetov z ON p.zoznam_predmet_id = z.zoznam_predmet_id
    WHERE r.rozvrh_id = OLD.id_rozvrh;
    SET sprava = CONCAT('Prítomný na predmete ', skratkaPredmetu);
    INSERT INTO oznamenia(osoba_id, datum, sprava)
    VALUES (OLD.ziak_id, NOW(), sprava);
END;

        foreign key (ziak_id, predmet_id) references ziak_predmety (ziak_id, predmet_id),

create definer = root@`%` trigger after_znamky_ziakov_insert
    after insert
    on znamky_ziakov
    for each row
BEGIN
    declare predmet VARCHAR(255);
    declare sprava VARCHAR(255);
    declare znamka_z_vypoctu int;
    select zp.znamka into znamka_z_vypoctu from znamka_z_percenta zp
            where zp.kategoria_id = new.kategoria_id and zp.od <= new.znamka and zp.do >= new.znamka;
    select z.nazov into predmet from predmety p join zoznam_predmetov z on z.zoznam_predmet_id = p.zoznam_predmet_id
             where p.predmet_id = new.predmet_id;

    if znamka_z_vypoctu is NULL then
        set sprava = concat('Pridaná známka v ', predmet, ' > ', new.znamka);
    else
        set sprava = concat('Pridaná známka v ', predmet, ' > ', new.znamka, '(', znamka_z_vypoctu ,')');
    end if;

    INSERT INTO oznamenia(osoba_id, datum, sprava)
    VALUES (NEW.ziak_id, NOW(), sprava);
END;

create definer = root@`%` trigger after_znamky_ziakov_update
    after update
    on znamky_ziakov
    for each row
BEGIN
    declare predmet VARCHAR(255);
    declare sprava VARCHAR(255);
    declare znamka_z_vypoctu_old int;
    declare znamka_z_vypoctu_new int;
    select zp.znamka into znamka_z_vypoctu_old from znamka_z_percenta zp
            where zp.kategoria_id = old.kategoria_id and zp.od <= old.znamka and zp.do >= old.znamka;
    select zp.znamka into znamka_z_vypoctu_new from znamka_z_percenta zp
            where zp.kategoria_id = new.kategoria_id and zp.od <= new.znamka and zp.do >= new.znamka;
    select z.nazov into predmet from predmety p join zoznam_predmetov z on z.zoznam_predmet_id = p.zoznam_predmet_id
             where p.predmet_id = old.predmet_id;

    if znamka_z_vypoctu_old is NULL then
        set sprava = concat('Zmenená známka v ', predmet, ' z ', old.znamka, ' na ', new.znamka);
    else
        set sprava = concat('Zmenená známka v ', predmet, ' z ', old.znamka, '(', znamka_z_vypoctu_old ,')', ' na ', new.znamka, '(', znamka_z_vypoctu_new ,')');
    end if;
    INSERT INTO oznamenia(osoba_id, datum, sprava)
    VALUES (old.ziak_id, NOW(), sprava);
END;
